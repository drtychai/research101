\fr{}{
    \Large{
        \begin{center}
            Before we think about research ideas,\vspace{1em}

            let's discuss what the \textit{ideal} one looks like
        \end{center}
    }
}

\begin{frame}{Characteristics of the Ideal Research Topic}
  %The characteristics of an ideal topic are to some extent incompatible:}%
    \begin{center}
        \vspace{-2em}
        \only<1>{
            \Large{\textbf{The subject should be timely.}}\vspace{2em}

            \textit{Previous groundwork should leave your research problem ripe for completion,
            and it should be in an active area with potential for future work and
            employment.}
        }

        \only<2>{
            \Large{\textbf{Your work should lead to a well-defined set of
            results to which you can lay claim.}}\vspace{2em}
 
            \textit{Results can be significantly harder to produce if you don't have a
            group of people working on closely related problems with whom you
            can interact and share code.}%
        } 

        \only<3>{
            \Large{\textbf{The best topics show a high level of creativity -- and
            are often somewhat speculative.}}\vspace{2em}

            \textit{It's often unclear how an idea will develop, at first.}
        }

        \only<4>{
            \Large{\textbf{You should really enjoy the subject, and want to spend the next several weeks/months/years with it!}}\vspace{2em}

            \textit{An ideal project in a foreign subject is of no use without a willing person/group to help direct you.}
        }
    \end{center}

\end{frame}

\fr{}{
    \begin{itemize}
        \item\textbf{The subject should be timely.}\vspace{0.5em}

        \item\textbf{Your work should lead to a well-defined set of
        results to which you can lay claim.}\vspace{0.5em}
 
    \item\textbf{The best topics show a high level of creativity -- and
        are often somewhat speculative.}\vspace{0.5em}

    \item\textbf{You should really enjoy the subject, and want to
        spend the next several weeks/months/years with it!}\vspace{0.5em}
    \end{itemize}

    \centering{\vspace{1em}But\ldots these characteristics are, to some extent, incompatible}\only<1>{\vspace{1em}}
    

    \only<2>{
        \centering{\textit{Clearly some compromise is necessary here!}}%
    }
}
